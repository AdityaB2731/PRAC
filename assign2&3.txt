#include <iostream>
#include <vector>
#include <queue>
using namespace std;

struct Node {
    int v;   // vertex
    int g;   // cost so far
    int h;   // heuristic

    // priority queue comparator
    bool operator<(const Node &other) const {
        return (g + h) > (other.g + other.h); // min heap
    }
};

int main() {
    int n = 5;

    // Graph: (neighbor, cost)
    vector<pair<int,int>> adj[5];

    adj[0] = {{1, 2}, {2, 4}};
    adj[1] = {{2, 1}, {3, 7}};
    adj[2] = {{4, 3}};
    adj[3] = {{4, 1}};
    adj[4] = {};

    // Heuristic values (to goal node 4)
    int h[] = {7, 6, 2, 1, 0};

    int start = 0;
    int goal = 4;

    priority_queue<Node> pq;
    pq.push({start, 0, h[start]});

    cout << "A* Traversal Path:\n";

    while (!pq.empty()) {
        Node cur = pq.top();
        pq.pop();

        cout << cur.v << " ";

        // Goal reached
        if (cur.v == goal) {
            cout << "\nGoal Reached!\n";
            break;
        }

        // Explore neighbors
        for (auto x : adj[cur.v]) {
            int next = x.first;
            int cost = x.second;

            pq.push({next, cur.g + cost, h[next]});
        }
    }

    return 0;
}

#include <iostream>
using namespace std;

int board[20][20];
int N;

// Check if safe to place queen
bool isSafe(int row, int col) {

    // Check column
    for(int i = 0; i < row; i++) {
        if(board[i][col] == 1)
            return false;
    }

    // Check left diagonal
    for(int i = row-1, j = col-1; i >= 0 && j >= 0; i--, j--) {
        if(board[i][j] == 1)
            return false;
    }

    // Check right diagonal
    for(int i = row-1, j = col+1; i >= 0 && j < N; i--, j++) {
        if(board[i][j] == 1)
            return false;
    }

    return true;
}

// Backtracking function
bool solve(int row) {

    if(row == N)
        return true;

    for(int col = 0; col < N; col++) {

        if(isSafe(row, col)) {

            board[row][col] = 1;

            if(solve(row + 1))
                return true;

            // Backtrack
            board[row][col] = 0;
        }
    }

    return false;
}

// Display board
void display() {
    cout << "\nN-Queens Solution:\n";
    for(int i = 0; i < N; i++) {
        for(int j = 0; j < N; j++) {
            if(board[i][j] == 1)
                cout << " Q ";
            else
                cout << " . ";
        }
        cout << endl;
    }
}

int main() {
    cout << "Enter value of N: ";
    cin >> N;

    // Handle invalid cases
    if(N == 2 || N == 3) {
        cout << "No solution exists for N = " << N << endl;
        return 0;
    }

    if(N == 1) {
        cout << " Q " << endl;
        return 0;
    }

    if(solve(0))
        display();
    else
        cout << "No solution exists!\n";

    return 0;
}
