1. Selection Sort (Very Easy Greedy)

👉 Idea: pick minimum element each time

#include <iostream>
using namespace std;

int main() {
    int n;
    cout << "Enter size: ";
    cin >> n;

    int a[100];
    for(int i = 0; i < n; i++)
        cin >> a[i];

    for(int i = 0; i < n-1; i++) {
        int min = i;

        for(int j = i+1; j < n; j++) {
            if(a[j] < a[min])
                min = j;
        }

        swap(a[i], a[min]);
    }

    cout << "Sorted array:\n";
    for(int i = 0; i < n; i++)
        cout << a[i] << " ";
}
✅ 2. Dijkstra (Shortest Path – Greedy)

👉 Idea: pick minimum distance node

#include <iostream>
using namespace std;

int main() {
    int n = 5;

    int graph[5][5] = {
        {0,2,0,1,0},
        {2,0,3,2,0},
        {0,3,0,0,1},
        {1,2,0,0,1},
        {0,0,1,1,0}
    };

    int dist[5], visited[5] = {0};

    for(int i = 0; i < n; i++)
        dist[i] = 999;

    dist[0] = 0;

    for(int i = 0; i < n; i++) {

        int u = -1;

        for(int j = 0; j < n; j++) {
            if(!visited[j] && (u == -1 || dist[j] < dist[u]))
                u = j;
        }

        visited[u] = 1;

        for(int v = 0; v < n; v++) {
            if(graph[u][v] && dist[u] + graph[u][v] < dist[v]) {
                dist[v] = dist[u] + graph[u][v];
            }
        }
    }

    cout << "Shortest distances:\n";
    for(int i = 0; i < n; i++)
        cout << dist[i] << " ";
}
✅ 3. Kruskal (Minimum Spanning Tree)

👉 Idea: pick smallest edge without cycle

#include <iostream>
#include <algorithm>
using namespace std;

struct Edge {
    int u, v, w;
};

bool cmp(Edge a, Edge b) {
    return a.w < b.w;
}

int parent[100];

int find(int x) {
    if(parent[x] == x)
        return x;
    return parent[x] = find(parent[x]);
}

void unite(int a, int b) {
    parent[find(a)] = find(b);
}

int main() {

    Edge edges[] = {
        {0,1,1},
        {1,2,2},
        {0,2,3}
    };

    int n = 3;

    for(int i = 0; i < n; i++)
        parent[i] = i;

    sort(edges, edges+3, cmp);

    cout << "MST Edges:\n";

    for(auto e : edges) {
        if(find(e.u) != find(e.v)) {
            cout << e.u << " - " << e.v << endl;
            unite(e.u, e.v);
        }
    }
}
